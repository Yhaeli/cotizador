// Poner automáticamente la fecha real de hoy al abrir
function iniciarFecha() {
    const hoy = new Date();
    const d = String(hoy.getDate()).padStart(2, '0');
    const m = String(hoy.getMonth() + 1).padStart(2, '0');
    const a = hoy.getFullYear();
    document.getElementById('img-fecha').innerText = `${d}/${m}/${a}`;
}

// Lógica inteligente para leer el cuadro de texto y calcular todo
function procesarCotizacion() {
    const texto = document.getElementById('pedido-texto').value;
    const lineas = texto.split('\n');
    const tablaFilas = document.getElementById('img-tabla-filas');
    
    tablaFilas.innerHTML = ""; // Limpiar tabla de la imagen
    let sumaPlantas = 0;
    let textoWhatsAppDetalle = "";
    let itemNum = 1;

    lineas.forEach(linea => {
        if(linea.trim() === "") return;

        let cantidad = 1;
        let descripcion = linea.trim();
        let precioUnitario = 0;

        // 1. Intentar buscar cantidad al inicio (ej: "2 Alocasia")
        const regexCantidad = linea.match(/^(\d+)\s+(.*)/);
        if (regexCantidad) {
            cantidad = parseInt(regexCantidad[1]) || 1;
            descripcion = regexCantidad[2];
        }

        // 2. Extraer el precio al final separándolo por un guion "-"
        if (descripcion.includes('-')) {
            const partes = descripcion.split('-');
            let precioStr = partes.pop().replace(/s\/\.?/gi, '').trim();
            precioUnitario = parseFloat(precioStr) || 0;
            descripcion = partes.join('-').trim();
        } else {
            // Si no tiene guion busca el último número suelto al final
            const regexPrecio = descripcion.match(/[\s\/.]+(\d+(?:\.\d+)?)$/);
            if(regexPrecio) {
                precioUnitario = parseFloat(regexPrecio[1]) || 0;
                descripcion = descripcion.substring(0, descripcion.lastIndexOf(regexPrecio[1])).replace(/s\/\.?/gi, '').trim();
            }
        }

        // Limpieza estética de símbolos que queden sueltos
        descripcion = descripcion.replace(/^[-–—:\s\./]+|[-–—:\s\./]+$/g, '').trim();
        if(!descripcion) descripcion = "Planta Especie";

        const totalFila = cantidad * precioUnitario;
        sumaPlantas += totalFila;

        // Insertar fila limpia en la tabla gráfica para la foto
        tablaFilas.innerHTML += `
            <tr>
                <td class="text-center">${cantidad}</td>
                <td>${descripcion}</td>
                <td class="text-right">S/. ${precioUnitario.toFixed(2)}</td>
                <td class="text-right">S/. ${totalFila.toFixed(2)}</td>
            </tr>
        `;

        // Añadir a la cadena de texto de WhatsApp
        textoWhatsAppDetalle += `${itemNum}. ${cantidad} Unid. de ${descripcion} - S/. ${precioUnitario.toFixed(2)} c/u\n`;
        itemNum++;
    });

    // Actualizar campos numéricos del formulario
    document.getElementById('valPlantas').value = sumaPlantas.toFixed(2);
    const embalaje = parseFloat(document.getElementById('valEmbalaje').value) || 0;
    const totalGeneral = sumaPlantas + embalaje;
    document.getElementById('valTotal').value = totalGeneral.toFixed(2);

    // Actualizar elementos visuales en el diseño del recibo
    const clienteAct = document.getElementById('cliente').value || '---';
    document.getElementById('img-cliente').innerText = clienteAct;
    document.getElementById('img-subtotal').innerText = sumaPlantas.toFixed(2);
    document.getElementById('img-embalaje').innerText = embalaje.toFixed(2);
    document.getElementById('img-total').innerText = totalGeneral.toFixed(2);

    // Preparar el formato de texto oculto para copiar a WhatsApp
    const fechaActual = document.getElementById('img-fecha').innerText;
    document.getElementById('output-texto').value = 
        `🪴 *COTIZACIÓN VIVERO ELIEL*\n\n` +
        `👤 *Cliente:* ${clienteAct}\n` +
        `📅 *Fecha:* ${fechaActual}\n` +
        `⏳ *Vigencia:* 24 horas\n\n` +
        `*Detalle del pedido:*\n${textoWhatsAppDetalle}` +
        `-------------------------------------------\n` +
        `🪴 *Total de plantas:* S/. ${sumaPlantas.toFixed(2)}\n` +
        `📦 *Embalaje y traslado:* S/. ${embalaje.toFixed(2)}\n` +
        `-------------------------------------------\n` +
        `💰 *TOTAL A CANCELAR: S/. ${totalGeneral.toFixed(2)}*\n\n` +
        `🪴 *Métodos de pago:*\n` +
        `📱 Yape: 981 046 861\n` +
        `🏦 BCP | Cta Corriente: 19178952536034\n` +
        `* CCI: 00219117895253603450\n` +
        `💥 Titular: Arlene Cruzado Llanos\n\n` +
        `⚠️ *Importante:* El flete es cobrado en destino por la agencia elegida (SHALOM/OLVA).`;
}

function guardarImagen() {
    procesarCotizacion();
    const contenedor = document.getElementById('bloque-imagen');
    const cliente = document.getElementById('cliente').value || 'Cliente';
    
    // Generar la imagen recortada justa y limpia sin bordes extras del navegador
    html2canvas(contenedor, { 
        scale: 2, // Mantiene la nitidez alta
        useCORS: true
    }).then(canvas => {
        const enlace = document.createElement('a');
        enlace.download = `Cotizacion_${cliente.replace(/\s+/g, '_')}.png`;
        enlace.href = canvas.toDataURL('image/png');
        enlace.click();
    });
}

function copiarWhatsApp() {
    procesarCotizacion();
    const tArea = document.getElementById('output-texto');
    tArea.select();
    tArea.setSelectionRange(0, 99999); // Soporte móvil
    
    try {
        navigator.clipboard.writeText(tArea.value).then(() => {
            alert('📋 ¡Texto copiado listo para pegar en WhatsApp!');
        }).catch(() => {
            document.execCommand('copy');
            alert('📋 ¡Copiado con éxito!');
        });
    } catch (e) {
        alert('Hubo un problema al copiar. Selecciona el texto manualmente.');
    }
}
